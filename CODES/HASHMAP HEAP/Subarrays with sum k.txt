class Solution
{
    static int findSubArraySum(int arr[], int N, int k)
    {
        // code here
        int ans = 0;
        int sum = 0;
        HashMap<Integer,Integer> h = new HashMap<>();
        h.put(0,1);
        
        for(int val:arr){
            sum+=val;
            
            int temp = sum - k;
            if(h.containsKey(temp))
                ans +=h.get(temp);
            h.put(sum,h.getOrDefault(sum,0)+1);		// default h yh toh
        
    }
        return ans;
}
        
    }
